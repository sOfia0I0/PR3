#include <iostream>
using namespace std;
int main(void)
{
    setlocale(LC_ALL, "Rus");
    int n;
    double p, sum = 0;
    cout << "Введите натуральное число n: ";
    cin >> n;
    cout << "Введите значение p: ";
    cin >> p;
    cout << "Введите " << n << " вещественных чисел (через точку):" << endl;
    for (int i = 0; i < n; i++)
    {
        double a;
        cin >> a;
        if (a > p)
        {
            sum += a;
        }
    }
    cout << "Сумма чисел, больших чем " << p << ", составляет: " << sum << endl;
    return 0;
}
